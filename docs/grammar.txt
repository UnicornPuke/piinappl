atom : if-expr
     : for-expr
     : while-expr
     : var-expr
     : loop-expr
     : func-def
     : expr

if-expr : KEYWORD:if COLON LPAREN expr LBRACE expr RBRACE RPAREN
          (KEYWORD:elif COLON LPAREN expr LBRACE expr RBRACE RPAREN)*
          (KEYWORD:else LBRACE expr RBRACE)?

var-expr : (KEYWORD:def|KEYWORD:manip) COLON LPAREN IDENTIFIER EQ expr RPAREN

loop-expr : KEYWORD:loop COLON LPAREN expr LBRACE atom RBRACE RPAREN

while-expr : KEYWORD:while COLON LPAREN expr LBRACE atom RBRACE RPAREN

for-expr : KEYWORD:for COLON LPAREN IDENTIFIER KEYWORD:in expr KEYWORD:to expr (KEYWORD:step expr)? LBRACE atom RBRACE RPAREN

func-def : KEYWORD:func COLON LPAREN IDENTIFIER? LPAREN (IDENTIFIER (COMMA IDENTIFIER)*)? RPAREN LBRACE expr RBRACE RPAREN

expr : comp-expr ((KEYWORD:and|KEYWORD:or) comp-expr)*

comp-expr : KEYWORD:not comp-expr
          : arith-expr ((EE|LT|GTE|LTE|GT) arith-expr)*

arith-expr : term ((PLUS|MINUS) term)*

term : power ((MUL|DIV) power)*

power : call (POWER call)*

call : factor (LPAREN (expr (COMMA expr)*)? RPAREN)?

factor : (PLUS|MINUS) factor
       : LPAREN expr RPAREN
       : list-call

list-call : list-expr (TT_PIPE list-expr)*

list-expr : LSQAURE (expr (COMMA expr)*)? RPAREN
          : STRING
